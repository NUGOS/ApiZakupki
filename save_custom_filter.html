POST remote/api/v1/save_filter

сохранить поисковый фильтр

UPDATE 2018-05-24

add entity_type

request headers:

  • Api-Agent - type: string, required
  • Acc-Token - type: string, required

possible request arguments:

  • filter_name - type: string, required
  • locations - type: array of string, possible values - one or more locations.value
  • use_delivery_location - type: bool
  • primary_classifier_id - type: array of int, possible values - one or more primary_classifier.value
  • amount_from - type: float
  • amount_to - type: float
  • merchant_name - type: array of string, possible values - one or more merchant_suggest.value
  • method_type - type: array of int, possible values - one or more method_type.value
  • range_date_created_dt_end - type: string, values format DD.MM.YYYY
  • range_date_tendering_start_dt_start - type: string, values format DD.MM.YYYY
  • range_date_tendering_start_dt_end - type: string, values format DD.MM.YYYY
  • range_date_tendering_end_dt_start - type: type: string, values format DD.MM.YYYY
  • range_date_tendering_end_dt_end - type: type: string, values format DD.MM.YYYY
  • range_date_auction_start_dt_start - type: type: string, values format DD.MM.YYYY
  • range_date_auction_start_dt_end - type: string, values format DD.MM.YYYY
  • status - type: array of int, possible values - one or more status.value
  • entity_type - type: array of int, possible values - one or more entity_type.value
  • assignee_employee - type: array of int, possible values - one or more idsassignment_data.available_users.id

possible response fields:

  • status - type: str, values: ok/error
  • errors - type: array of objects, validation errors
  • data - type: object
  • system - type: str, system error messages

possible response data fields:

  • status - type: array of status objects or empty, possible values - one or more stauses, described below
  • method_type - type: array method_type objects or empty, possible values - one or more method_types, described below
  • range_date_auction_start_dt_end - type: date str or null, format DD.MM.YYYY
  • range_date_tendering_end_dt_start - type: date str or null, format DD.MM.YYYY
  • locations - type: array of location objects or empty, possible values - one or more locations, described below
  • range_date_tendering_end_dt_end - type: date str or null, format DD.MM.YYYY
  • amount_from - type: float str or null
  • amount_to - type: float str or null
  • range_date_created_dt_start - type: date str or null, format DD.MM.YYYY
  • range_date_created_dt_end - type: date str or null, format DD.MM.YYYY
  • keywords - type: array of str or empty
  • filter_id - type: int (signed)
  • range_date_auction_start_dt_start - type: date str or null, format DD.MM.YYYY
  • range_date_tendering_start_dt_end - type: date str or null, format DD.MM.YYYY
  • primary_classifier_id - type: array of primary_classifier_id objects or empty, described below
  • classifier_id - type: array, always empty
  • range_date_tendering_start_dt_start - type: date str or null, format DD.MM.YYYY
  • merchant_name - type: array of merchant_name objects or empty
  • entity_type - type: array of entity_type objects, described below

status object fields:

  • value_label - type: str
  • value - type: int
  • label - type: str

method_type object fields:

  • value_label - type: str
  • value - type: int
  • label - type: str

location object fields:

  • id - type: int
  • value - type: str
  • label - type: str

merchant_name object fields:

  • text - type: str
  • srn - type: str
  • value - type: str

primary_classifier_id object fields:

  • full_title - type: text
  • code - type: text
  • value - type: int
  • title - type: text

entity_type object fields:

  • value - type: int
  • value_label - type: str
  • label - type: str

NOTE: во время написания документации пришло понимание, что у нас есть дублирующиеся данные а именно - amount и amount_form/amount_to, есть подозрение, что что-то из этого явно лишнее и можно от этого избавится, но, в рамках этой задачи реализовываться это не будет мы сохраним данный набор полей, для поддержания работы api и совместимости с приложение но, когда будет производить рефакторинг учтем этот нюанс и предупредим в случаях изменений

DELETE remote/api/v1/delete_filter/{filter_id}

сохранить поисковый фильтр filter_id - id сохраненного фильтра, можно получить из предыдущего метода или же тут

request headers:

  • Acc-Token - type: string, required

possible response fields:

  • status - type: str, values: ok/error
  • errors - type: array of objects, validation errors
  • data - type: object
  • system - type: str, system error messages

при успешном выполнении запроса всегда возвращается пустой ответ

curl --header "Acc-Token: eyJlbXBsb3llZSI6NjEyNjEsIm1lcmNoYW50Ijo1NTEzOH0.C8di6Q.JPtyyp32oH7KaM7WqyjQPE0SoXA" -X DELETE "http://my.example.com:5001/remote/api/v1/delete_filter/100500"
{
  "status": "ok",
  "errors": [
  ],
  "data": {
  },
  "system": ""
}

POST remote/api/v1/rename_filter/{filter_id}

переименовать фильтр filter_id - id сохраненного фильтра можно получить тут

request headers:

  • Acc-Token - type: string, required

possible request arguments:

  • filter_name - type: string, required

possible response fields:

  • status - type: str, values: ok/error
  • errors - type: array of objects, validation errors
  • data - type: object
  • system - type: str, system error messages

possible response data fields:

  • id - type: int, filter id
  • name - type: str, new filter name
// request body
{  
    "filter_name":"new awesome filter name"
}
curl --header "Acc-Token: eyJlbXBsb3llZSI6NjEyNjEsIm1lcmNoYW50Ijo1NTEzOH0.C8di6Q.JPtyyp32oH7KaM7WqyjQPE0SoXA" -X POST "http://my.example.com:5001/remote/api/v1/rename_filter/100500" -d @request_bodu.json
{
  "status": "ok",
  "errors": [
  ],
  "data": {
    "id":10050,
    "name": "new awesome filter name"
  },
  "system": ""
}
