GET remote/api/v1/entity

GET remote/api/v2/entity

GET remote/api/v3/search

UPDATE 2018-05-11

entity_type filter param added, only v2

UPDATE 2019-02-26

id removed in V3

request headers:

  • Api-Agent - type: string, reuqired
  • Acc-Token - type: string, required

possible request arguments:

  • page - type: int, if not provided equals to 1, single
  • saved_filter_id - type: int, value - one of id from saved_filter list returned in auth response, single
  • mode - type: str, one of ["all", "starred", "planned", "participated", "won", "supplier_recommended"], single
  • search - type: str, search phrase, single
  • locations - type: str, value - one of ["95-98", "01-06", "76-78", "40-42", "83-87", "07-09", "65-68", "21-24", "43-45", "91-94", "79-82", "33-35", "36-39", "73-75", "18-20", "10-13", "69-72", "99", "61-64", "29-32", "58-60", "88-90", "54-57", "14-17", "46-48", "25-28", "49-53"], multiply
  • use_delivery_location - type: str, value - one of ["true", "false"], single
  • primary_classifier_id - type: int, value - one of CPV classifier id, multiply
  • classifier_id - type: int, value - one of DKPP classifier id, multiply
  • amount_from - type: float, single
  • amount_to - type: float, single
  • method_type - type: int, value - one of [0, 1, 2, 7, 3, 4, 5, 8, 9, 14], multiply
  • range_date_created_dt_start - type: date, format: dd.mm.yyyy, single
  • range_date_created_dt_end - type: date, format: dd.mm.yyyy, single
  • range_date_tendering_start_dt_start - type: date, format: dd.mm.yyyy, single
  • range_date_tendering_start_dt_end - type: date, format: dd.mm.yyyy, single
  • range_date_tendering_end_dt_start - type: date, format: dd.mm.yyyy, single
  • range_date_tendering_end_dt_end - type: date, format: dd.mm.yyyy, single
  • range_date_auction_start_dt_start - type: date, format: dd.mm.yyyy, single
  • range_date_auction_start_dt_end - type: date, format: dd.mm.yyyy, single
  • merchant_name - type: text, multiply
  • merchant_srn - type: text
  • status - type: int, values - one of [0, 1, 2, 10, 3, 4, 5, 6, 7, 12], multiply
  • is_blocked - type: bool
  • entity_type - type: int, values - one of [0, 2], by default or if not provided - both will be applied, not supported in v1, details
  • assignee_employee - type: int, values - see assignment_data

possible response fields:

  • status - type: str, values: ok/error
  • errors - type: array of objects, validation errors
  • data - type: object
  • system - type: str, system error messages

possible response data fields:

  • total_items - type: int, total found entities
  • items - type: array of item objects
  • page - type: int, current page

possible items object fields:

  • id - type: long removed in version 3 use uid instead
  • uid - type: string
  • status - type: int, one of [0, 1, 2, 10, 3, 4, 5, 6, 7, 12]
  • has_lots - type: boolean, describes is current item has lots or no
  • method_type - type: int, one of [0, 1, 2, 7, 3, 4, 5, 8, 9, 14, 16]
  • descr - type: str
  • title - type: str
  • enquiry_period - type: object, state period object, described below
  • auction_period - type: object, state period object, described below
  • tendering_period - type: object, state period object, described below
  • currency_iso_code - type: str, one of ["UAH", "USD", "EUR", "RUB", "GBP"]
  • amount - type: float
  • plan_to_participate - type: boolean
  • participate - type: boolean
  • date_created - type: str, format: yyyy-mm-ddTHH:MM:SS.fZ
  • status_text - type: str
  • tax_included - type: boolean
  • merchant_name - type: str
  • is_favorite - type: boolean
  • entity_type - type: int, values - one of [0, 2], details
  • assignee_supplier_employee - type: assignee supplier object or null, described below

assignee supplier object fields:

  • id - type: int
  • name - type: str
  • is_iam - type: bool

state period object fields:

  • start - type: str or null, format: yyyy-mm-ddTHH:MM:SS.fZ
  • end - type: str or null, format: yyyy-mm-ddTHH:MM:SS.fZ
